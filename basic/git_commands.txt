git - система контроля версий
github - сервис/платформа/сайт, предоставляющий хранилище для версий кода


git --version - проверка версий
git config --global user.email "d.aksamaev@gmail.com" -
git config --global user.name "Donny87"

git config --list - проверили настройки

ssh-keygen - создали ключ ssh

cat ~/.ssh/id_rsa.pub - получение ssh ключа

создаем репозиторий на github

переходим в папку с нашим проектом
git init - превращает текущую директорию в git репозиторий
git remote add название связи(обычно - origin) ccskre_yf_elfktyysq_htgjpbnjhbq(SSH: git@github:..)
git remote -v - просмотр списков удаленных репозиториев
git status - проверка статуса файлов
git add название_файлов - отслеживание конкретных файлов
git add . - отслеживание всехafqkjd


git commit - создание коммита(версии/слепка кода)
git commit -m 'описание версии' - создание коммита с комментарием

git push название_связи(обычно origin) название_ветки - отправка версии кода на удаленный репозиторий
Например: git push origin master

git clone ссылка на удаленный репозиторий - склонировать  либо скопировать либо скачать удаленный репозиторий

git diff - просмотр внесенных изменений

git branch - вывод списка веток в терминал(звездочкой помеченна текущая ветка)
git branch название_ветки - создание новой название_ветки

git checkout название_ветки
git switch название_ветки  - переход на указанную ветку

git pull название_связи название_ветки - стягивание изменений с удаленного репозитория, с указанной название_ветки
